        public static class RemoteSession
    {

        private static final String TAG = RemoteSession.class.getName();
        private ReaderSession readerSession;

        public RemoteSession()
        {
        }

        public ApduResult process(ApduCommand command)
        {
            if (isOpen())
            {
                return readerSession.process(command);
            }
            Log.e(TAG, "ReaderSession is not open");
            return null;
        }

        public boolean open(MobileKeys mobileKeys)
        {
            if (isOpen())
            {
                try
                {
                    readerSession = mobileKeys.openReaderSession();
                    return true;
                }
                catch (MobileKeysException e)
                {
                    Log.e(TAG, "Error while opening remote session: ", e);
                }
            }
            Log.e(TAG, "Remote session is already open");
            return false;
        }

        public void close()
        {
            if (readerSession != null)
            {
                readerSession.close();
            }
        }

        public boolean isOpen()
        {
            return readerSession != null && readerSession.isSessionOpen();
        }
    }
